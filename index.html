Mini DJ Beat Game

Aperte A quando o bloco encostar na linha!

from js import document, window from pyodide.ffi import create_proxy import random canvas = document.getElementById("gameCanvas") ctx = canvas.getContext("2d") notes = [] score = 0 game_speed = 3 beat_line_y = 420 class Note: def __init__(self): self.x = 180 self.y = 0 self.hit = False def update(self): self.y += game_speed def draw(self): ctx.fillStyle = "cyan" ctx.fillRect(self.x, self.y, 40, 20) def spawn_note(): notes.append(Note()) def draw_beat_line(): ctx.fillStyle = "red" ctx.fillRect(0, beat_line_y, 400, 5) def draw_score(): ctx.fillStyle = "white" ctx.font = "20px Arial" ctx.fillText(f"Score: {score}", 10, 30) def update(): global score ctx.clearRect(0, 0, 400, 500) draw_beat_line() draw_score() for note in notes: note.update() note.draw() window.requestAnimationFrame(update_proxy) def keydown(event): global score if event.key.lower() == "a": for note in notes: if abs(note.y - beat_line_y) < 20 and not note.hit: score += 1 note.hit = True document.addEventListener("keydown", create_proxy(keydown)) # ---- LOOP CORRETO DE SPAWN ---- def spawn_loop(): spawn_note() window.setTimeout(spawn_proxy, 1000) spawn_proxy = create_proxy(spawn_loop) update_proxy = create_proxy(update) update() spawn_loop()
